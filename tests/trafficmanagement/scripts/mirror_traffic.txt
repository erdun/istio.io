set -e
set -u
set -o pipefail

source ${REPO_ROOT}/content/en/docs/tasks/traffic-management/mirroring/snips.sh

export SLEEP_POD=$(kubectl get pod -l app=sleep -o jsonpath={.items..metadata.name})

kubectl get pods
echo "SLEEP_POD: $SLEEP_POD"

# TODO: Should use snip_creating_a_default_routing_policy_2 here. Problem is it uses -it options.
kubectl exec "$SLEEP_POD" -c sleep -- curl -o /dev/null -s -w "%{http_code}\n" http://httpbin:8000/ISTIO_IO_MIRROR_TEST_1

echo "Done curl 1"

out=$(snip_creating_a_default_routing_policy_3 2>&1)
_verify_contains "$out" "ISTIO_IO_MIRROR_TEST_1" "snip_creating_a_default_routing_policy_3"

out=$(snip_creating_a_default_routing_policy_4 2>&1)
_verify_not_contains "$out" "ISTIO_IO_MIRROR_TEST_1" "snip_creating_a_default_routing_policy_4"

snip_mirroring_traffic_to_v2_1

# Wait a bit
sleep 10

# TODO: Should use snip_mirroring_traffic_to_v2_2 here. Problem is it uses -it options.
kubectl exec "$SLEEP_POD" -c sleep -- curl --retry 3 -o /dev/null -s -w "%{http_code}\n" http://httpbin:8000/ISTIO_IO_MIRROR_TEST_2

echo "Done curl 2"

out=$(snip_mirroring_traffic_to_v2_3 2>&1)
_verify_contains "$out" "ISTIO_IO_MIRROR_TEST_2" "snip_mirroring_traffic_to_v2_3"

out=$(snip_mirroring_traffic_to_v2_4 2>&1)
_verify_contains "$out" "ISTIO_IO_MIRROR_TEST_2" "snip_mirroring_traffic_to_v2_4"
